void bubble_sort(int *a, int size)
{
    int temp;
    int swap;
    while (1)
    {
        swap=0;
        for(int i=0; i<size-1;i++)
        {
            if(a[i]>a[i+1])
            {
                temp=a[i];
                a[i]=a[i+1];
                a[i+1]=temp;
                swap=1;
            }
        }
        if(swap==0)
        {break;}
    }
}
