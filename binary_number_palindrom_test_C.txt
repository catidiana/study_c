int bynary_palindrom (int a)
{
  int b=0;
  int test=a;
  while (test!=0) {
      int mod=test%2;
      b=b*2+mod;
      test=test/2;
  }
if (b==a) return 1; else return 0;
}
int bynary_palindrom1 (int a)
{
  int b=0;
  int test=a;
  while (test!=0) {
      int mod=test & 1;
      b=b<<1 | mod;
      test=test>>1;
  }
if (b==a) return 1; else return 0;
}
