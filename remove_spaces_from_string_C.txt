void compactString (char * str)
{
    int s=strlen(str);
    int j=0; int count=0;
    for (int i=0;i<s;i++) {
        if (str[i]!=' ')
        { str[j]=str[i];
            j=j+1;
        }
        else {
            str[j]=str[i+1];
            count=count+1;
        }
    }
    str[s-count]='\0';
    return;
}
